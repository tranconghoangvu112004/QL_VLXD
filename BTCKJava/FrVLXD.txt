    // load du lieu len bang
    public void loadBang(){
        try {
            conn = cn.getConnection();
            int number;
            Vector row;
            String sql = "select * from QLVLXD";
            Statement st = conn.createStatement();
            ResultSet rs = st.executeQuery(sql);
            ResultSetMetaData metadata = rs.getMetaData();
            number = metadata.getColumnCount();
            tb.setRowCount(0);
            while (rs.next()){
                row = new Vector();
                for(int i=1;i<=number;i++)
                    row.addElement(rs.getString(i));
                    tb.addRow(row);
                tblVLXD.setModel(tb);
                }
            st.close();
            rs.close();
            conn.close();
        } catch (Exception e) {
        }
        //hien du lieu bang len txt
        tblVLXD.getSelectionModel().addListSelectionListener(new ListSelectionListener() {
            @Override
            public void valueChanged(ListSelectionEvent e) {
                if(tblVLXD.getSelectedRow()>=0){
                    txtID.setText(tblVLXD.getValueAt(tblVLXD.getSelectedRow(), 0)+"");
                    txtTen.setText(tblVLXD.getValueAt(tblVLXD.getSelectedRow(), 1)+"");
                    txtGiaban.setText(tblVLXD.getValueAt(tblVLXD.getSelectedRow(), 2)+"");
                    txtSoLuong.setText(tblVLXD.getValueAt(tblVLXD.getSelectedRow(), 3)+"");
                    txtDate.setText(tblVLXD.getValueAt(tblVLXD.getSelectedRow(), 4)+"");
                    }
        }
    });
    }

    private void btnFindActionPerformed(java.awt.event.ActionEvent evt) {                                        
        if(txtTimTenVLXd.getText().equals(""))
            loadBang();
        else{
            String search = txtTimTenVLXd.getText().replaceAll("[-'!]","");
            String sql =  "select * from QLVLXD where Ten_VLXD like '%" + txtTimTenVLXd.getText() + "%'";
            conn = cn.getConnection();
            PreparedStatement pst;
        try { 
            pst = conn.prepareStatement(sql);
            ResultSet rs = pst.executeQuery();
            tb.setRowCount(0);
            while(rs.next()){
                Object personObject[]= {
                    rs.getString("ID"),
                    rs.getString("Ten_VLXD"),
                    rs.getString("Gia_ban"),
                    rs.getString("So_luong_da_ban"),
                    rs.getString("Ngay_ban")
                };
                tb.addRow(personObject);
                tblVLXD.setModel(tb);
                tblVLXD.getColumnModel().getColumn(1).setPreferredWidth(50);
            }
        }catch (Exception e){
        }    
        }
    }

    private void btnAddActionPerformed(java.awt.event.ActionEvent evt) {                                       
          conn = cn.getConnection();
          try {
            if(txtID.getText().equals("")||txtTen.getText().equals("")||txtGiaban.getText().equals("")||txtSoLuong.getText().equals("")||txtDate.getText().equals("")){
                JOptionPane.showMessageDialog(this, "Bạn cần nhập đủ dữ liệu");
            }else {
                StringBuffer sb = new StringBuffer();
                String sql_check_pk = "select ID from QLVLXD where ID = '"+txtID.getText()+"'";
                Statement st = conn.createStatement();
                rs = st.executeQuery(sql_check_pk);
                if(rs.next()){
                    sb.append("ID đã tồn tại ! Vui lòng nhập lại!");
                }
                if(sb.length()>0){
                    JOptionPane.showConfirmDialog(this, sb.toString());
                }else{
                    String sql = "insert into QLVLXD(ID,Ten_VLXD,Gia_ban,So_luong_da_ban,Ngay_ban) values ('"+txtID.getText()+"','"+txtTen.getText()+"','"+txtGiaban.getText()+"','"+txtSoLuong.getText()+"','"+txtDate.getText()+"')";
                    st = conn.createStatement();
                    int kq = st.executeUpdate(sql);
                    if(kq>0){
                        JOptionPane.showMessageDialog(this, "Thêm mới thành công");
                        xoaTrang();
                    }
                }
                st.close();
            }
            conn.close();
            rs.close();
        } catch (Exception e) {
        }
    }

    private void btnEditActionPerformed(java.awt.event.ActionEvent evt) {
